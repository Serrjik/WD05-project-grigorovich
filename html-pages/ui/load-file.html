Для использования миксина кнопки загрузки файла в него передаются параметры вида "Заголовок","Описание".
Если нужна скруглённая кнопка - дополнительно передается третий параметр "radius".
Если нужна кнопка с картинкой - четвёртым параметром передаём "image". Например:

+loadFile("Заголовок", "Описание")

+loadFile("Заголовок", "Описание", "radius")

+loadFile("Заголовок", "Описание", "", "image")

+loadFile("Заголовок", "Описание", "radius", "image")

mixin loadFile(title, opis, type, image)
	if type && image 
		.load-file-title Загрузка файлов - UI-kit и HTML фреймворк - Документация		
		.load-file-opis 
		.load-file-fieldset
			input#file.inputfile.inputfile-rad(type='file', name='file', data-multiple-caption='{count} файлов выбрано', multiple='')
			label(for='file') Выбрать файл
			span Файл не выбран
		.load-file-wrap
			.load-file-wrap-img
				img(src="../img/uploadfiles/loadfile.jpg", alt="foto").load-file-wrap-img__image
			.load-file-wrap__button
				+buttonDelete('Удалить', '#!', 'small')
	else if type
		.load-file-title Загрузка файлов - UI-kit и HTML фреймворк - Документация		
		.load-file-opis 
		.load-file-fieldset
			input#file.inputfile.inputfile-rad(type='file', name='file', data-multiple-caption='{count} файлов выбрано', multiple='')
			label(for='file') Выбрать файл
			span Файл не выбран
	else if image
		.load-file-title Загрузка файлов - UI-kit и HTML фреймворк - Документация		
		.load-file-opis 
		.load-file-fieldset
			input#file-2.inputfile(type='file', name='file-2', data-multiple-caption='{count} файлов выбрано',multiple='')
			label(for='file-2') Выбрать файл
			span Файл не выбран
		.load-file-wrap
			.load-file-wrap-img
				img(src="../img/uploadfiles/loadfile.jpg", alt="foto").load-file-wrap-img__image
			.load-file-wrap__button
				+buttonDelete('Удалить', '#!', 'small')
	else
		.load-file-title Загрузка файлов - UI-kit и HTML фреймворк - Документация
